% ── Override \@chapter: use fancyhdr, add chaptermark ────────────
\def\@chapter[#1]#2{\ifnum \c@secnumdepth >\m@ne
    \refstepcounter{chapter}
    \typeout{\@chapapp\space\thechapter:}
    \addcontentsline{toc}{chapter}{\@chapapp\space\thechapter:\hspace*{1em}#1}\else
    \addcontentsline{toc}{chapter}{#1}\fi
    \addtocontents{lof}{\protect\addvspace{10pt}}
    \addtocontents{lot}{\protect\addvspace{10pt}}
    \@makechapterhead{#2} \@afterheading
    \ifnum\chap@or@app=1 \ifnum\value{chapter}=1
        \markboth{}{}\pagestyle{fancy}%
        \ifnum\with@parts=0 \fi
    \fi\fi
    \chaptermark{#1}%
    \thispagestyle{plain}}

% ── Part pages: gray number, small-caps title, centered ──────────
\def\@part[#1]#2{%
    \short@page
    \thispagestyle{plain}%
    \ifnum \c@secnumdepth >-2 \relax
      \refstepcounter{part}%
      \addcontentsline{toc}{part}{\partname~\thepart\hspace{1em}#1}%
    \else
      \addcontentsline{toc}{part}{#1}%
    \fi
    \markboth{}{}%
    {\centering
     \interlinepenalty \@M
     \reset@font
     \ifnum \c@secnumdepth >-2 \relax
       {\fontsize{40}{48}\selectfont\color{chaptergray} \partname~\thepart}%
       \par
       \vskip 20\p@
     \fi
     {\Huge\scshape #2}\par}%
    \ifnum\value{part}=1
     \with@parts=1 \fi
    \@endpart}
