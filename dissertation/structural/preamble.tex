% Parts: bold, slightly larger, no page number, generous space above
\renewcommand*\l@part[2]{%
  \ifnum \c@tocdepth >-2\relax
    \addpenalty{-\@highpenalty}%
    \addvspace{1.8em \@plus\p@}%
    \begingroup
      \parindent \z@
      \leavevmode
      {\large \bfseries #1}\par
      \nobreak
      \vskip 4pt
      \global\@nobreaktrue
      \everypar{\global\@nobreakfalse\everypar{}}%
    \endgroup
  \fi}

% Chapters: bold, normal size, moderate space above
\def\l@chapter#1#2{\addpenalty{-\@highpenalty}
    \vskip 0.6em plus 1pt
    \@tempdima=\toc@chap@indent
    \begingroup
    \parindent \z@
    \rightskip \@pnumwidth
    \parfillskip -\@pnumwidth
    \leavevmode \bfseries
    \advance\leftskip\@tempdima \hskip -\leftskip #1\nobreak\hfill
    \nobreak\hbox to\@pnumwidth{\hss #2}\par
    \penalty\@highpenalty \endgroup}
