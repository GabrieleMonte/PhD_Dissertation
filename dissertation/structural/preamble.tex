% ── ToC entry formatting ──────────────────────────────────────────
% Parts: bold, slightly larger, no page number, generous space above
\renewcommand*\l@part[2]{%
  \ifnum \c@tocdepth >-2\relax
    \addpenalty{-\@highpenalty}%
    \addvspace{1.8em \@plus\p@}%
    \begingroup
      \parindent \z@
      \leavevmode
      {\large \bfseries #1}\par
      \nobreak
      \vskip 4pt
      \global\@nobreaktrue
      \everypar{\global\@nobreakfalse\everypar{}}%
    \endgroup
  \fi}

% Chapters: bold, normal size, moderate space above
\def\l@chapter#1#2{\addpenalty{-\@highpenalty}
    \vskip 0.6em plus 1pt
    \@tempdima=\toc@chap@indent
    \begingroup
    \parindent \z@
    \rightskip \@pnumwidth
    \parfillskip -\@pnumwidth
    \leavevmode \bfseries
    \advance\leftskip\@tempdima \hskip -\leftskip #1\nobreak\hfill
    \nobreak\hbox to\@pnumwidth{\hss #2}\par
    \penalty\@highpenalty \endgroup}

% ── Consistent front-matter headings (JCAP style) ────────────────

\def\utabstract{\clearpage
    \short@page
    \setcounter{regular@short}{1}
    \pagestyle{plain}
    \shortchapter*{Abstract}
    \default@spacing}

\def\tableofcontents{%
    \shortchapter*{Table of Contents}
    \pagestyle{plain}
    \baselineskip=16pt plus .5pt minus .5pt
    \hrule\vskip6pt
    \@starttoc{toc}%
    \vskip6pt\hrule}

% LoF with thin horizontal rules (not added to ToC)

    \shortchapter*{List of Figures}
    \pagestyle{plain}
    \baselineskip=14pt plus .5pt minus .5pt
    \hrule\vskip6pt
    \@starttoc{lof}%
    \vskip6pt\hrule}

% LoT with thin horizontal rules (not added to ToC)
\def\listoftables{%

    \pagestyle{plain}
    \baselineskip=14pt plus .5pt minus .5pt
    \hrule\vskip6pt
    \@starttoc{lot}%
    \vskip6pt\hrule}
